.text
.globl main

main:
    # GPIO base = 0x1000
    lui   x3, 0x1

    # x7 = 5000 (para delay)
    lui   x7, 0x1
    addi  x7, x7, 904

loop:
    lw    x5, 0(x3)      # leer switches (según tu GPIO)
    sw    x5, 0(x3)      # escribir LEDs/7seg (según tu GPIO)

    addi  x6, x0, 0
delay:
    addi  x6, x6, 1
    bne   x6, x7, delay

    beq   x0, x0, loop   # salto infinito
